TRACE
(implies) = define
  (p) = take
  (q) = take
  p = take (p)
  conjure (q) q using p

(iff) = define
  (p) = take
  (q) = take
  (p_implies) = give (p) to (implies)
  (p_q) = give (q) to (p_implies)
  (q_implies) = give (q) as (p) to (implies)
  (q_p) = give (p) as (q) to (q_implies)
  conjure (p_q) p_q
  conjure (q_p) q_p

(false) = define
  (p) = take
  conjure (p) p

(not) = define
  (p) = take
  p = take (p)
  conjure (false) false using p

not_apply = define
  TRACE
  (P) = take
  (not_P) = give (P) as (p) to (not)
  not_P = take (not_P)
  P = take (P)
  not_P_but_P = give P as p to not_P
  false = take from not_P_but_P
  give false

(not_not) = define
  (p) = take
  (not_p) = give (p) to (not)
  (not_not_p) = give (not_p) as (p) to (not)
  not_not_p = take (not_not_p)
  conjure (p) p using not_not_p

# axiom
not_not = take (not_not)

contraposition = define
  (p) = take
  (q) = take

  (p_implies) = give (p) to (implies)
  (p_q) = give (q) to (p_implies)

  (not_p) = give (p) to (not)
  (not_q) = give (q) as (p) to (not)

  p_q = take (p_q)
  not_q = take (not_q)

  not_p = define
    p = take (p)
    q_ = give p to p_q
    q = take from q_
    false_ = give q as p to not_q
    false = take from false_
    give false

  give not_p

contraposition_reverse = define
  TRACE

  (P) = take
  (Q) = take

  (not_P) = give (P) as (p) to (not)
  (not_Q) = give (Q) as (p) to (not)

  (contra_) = give (not_Q) as (p) to (implies)
  (contra) = give (not_P) as (q) to (contra_)

  (not_not_Q) = give (not_Q) as (p) to (not)

  contra = take (contra)
  P = take (P)

  not_Q_not_P = define
    not_Q = take (not_Q)
    contra = give not_Q as p to contra
    not_P = take q from contra
    give not_P

  not_Q_not_P_renamed = define
    p = take (not_Q)
    not_Q_not_P = give p as not_Q to not_Q_not_P
    q = take not_P from not_Q_not_P
    give q

  not_not_P = define
    not_P = take (not_P)
    not_P = give P as p to not_P
    false = take from not_P
    give false

  not_not_P_renamed = define
    p = take (not_P)
    not_not_P = give p as not_P to not_not_P
    false = take from not_not_P
    give false

  contraposition = give (not_Q) as (p) to contraposition
  contraposition = give (not_P) as (q) to contraposition

  contraposition = give not_Q_not_P_renamed as p_q to contraposition
  contraposition = give not_not_P_renamed as not_q to contraposition

  not_not_Q = take not_p from contraposition

  not_not = give (Q) as (p) to not_not
  not_not = give not_not_Q as not_not_p to not_not
  Q = take p from not_not

  give Q
