  define signature Not_P

  take p_q of signature P_Q

  give not_p

prove contraposition of signature Contraposition
  take signature P
  take signature Q

  define signature P_Q
    take p of signature P
    conjure q of signature Q
    give q

  define signature Not_P
    take p of signature P

    give false

  take p_q of signature P_Q
